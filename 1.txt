SELECT w.id, w.surname, w.name, w.patronymic, w.login, w.password, w.email,
	d.title as dept_title,
	(SELECT p1.title FROM post p1 WHERE p1.id = 1) AS post_title,
	(select d1.worker_id from dept d1 
		WHERE d1.title = d.title and d1.post_id = 1) AS worker
FROM worker w
LEFT JOIN dept d ON d.worker_id = w.id
WHERE w.id = 33;



# Создаем псевдонимы для таблиц
w = aliased(Worker)
d = aliased(Dept)
d1 = aliased(Dept)
p1 = aliased(Post)

# Создаем подзапросы
subquery_post_title = select([p1.title]).where(p1.id == 1).scalar_subquery()
subquery_worker = select([d1.worker_id]).where(d1.title == d.title).where(d1.post_id == 1).scalar_subquery()

# Создаем основной запрос
query = select([
    w.id,
    w.surname,
    w.name,
    w.patronymic,
    w.login,
    w.password,
    w.email,
    d.title.label('dept_title'),
    subquery_post_title.label('post_title'),
    subquery_worker.label('worker')
]).select_from(w).outerjoin(d, d.worker_id == w.id).where(w.id == 33)
